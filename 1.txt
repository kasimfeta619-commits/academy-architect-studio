<script>
    const canvas = document.getElementById('canvas-box');
    const ctx = canvas.getContext('2d');
    canvas.width = window.innerWidth / 2;
    canvas.height = window.innerHeight;

    let drawing = false, start = null;

    canvas.addEventListener('mousedown', e => {
        drawing = true;
        start = { x: e.offsetX, y: e.offsetY };
    });

    canvas.addEventListener('mouseup', e => {
        if (!drawing) return;
        const end = { x: e.offsetX, y: e.offsetY };
        
        // 2D чертане
        ctx.beginPath();
        ctx.moveTo(start.x, start.y);
        ctx.lineTo(end.x, end.y);
        ctx.strokeStyle = '#000';
        ctx.lineWidth = 5;
        ctx.stroke();

        // 3D симулация
        const dist = Math.sqrt((end.x-start.x)**2 + (end.y-start.y)**2).toFixed(0);
        document.getElementById('view-box').innerHTML = 
            `<div style="text-align:center;">
                <h3>Стена генерирана</h3>
                <p>Дължина: ${dist} см</p>
                <div style="width:${dist/2}px; height: 100px; background: #555; border: 1px solid #fff;"></div>
            </div>`;
        drawing = false;
    });
</script>
